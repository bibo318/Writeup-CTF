# Nmap Quiz 
1. help = -h
2. Syn Scan = -sS
3. UDP Scan = -sU
4. OS detection = -O
5. service verstion detection = -sV
6. verbosity flag = -v
7. very verbose = -vv
8. output in xml format = -oX
9. Aggressive scan = -A
10. set timing to max = -T5 (1-5)
11. specific port = -p
12. every port = -p-
13. use a script = --script
14. use script in vulnerability category = --script vuln
15. skip ping = -Pn
